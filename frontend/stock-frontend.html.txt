// ─── CANVAS CHART ─────────────────────────────────────────────────
function drawChart(prices) {
  const canvas = document.getElementById('price-chart');
  const ctx = canvas.getContext('2d');
  const dpr = window.devicePixelRatio || 1;
  const rect = canvas.parentElement.getBoundingClientRect();
  canvas.width = rect.width * dpr;
  canvas.height = 200 * dpr;
  canvas.style.width = rect.width + 'px';
  canvas.style.height = '200px';
  ctx.scale(dpr, dpr);

  const W = rect.width, H = 200, pad = 16;
  ctx.clearRect(0, 0, W, H);

  const min = Math.min(...prices), max = Math.max(...prices);
  const range = max - min || 1;
  const pts = prices.map((p, i) => ({
    x: pad + (i / (prices.length - 1)) * (W - pad * 2),
    y: H - pad - ((p - min) / range) * (H - pad * 2)
  }));

  // gradient area
  const grad = ctx.createLinearGradient(0, 0, 0, H);
  grad.addColorStop(0, 'rgba(0,212,170,0.25)');
  grad.addColorStop(1, 'rgba(0,212,170,0)');
  ctx.beginPath();
  ctx.moveTo(pts[0].x, H - pad);
  pts.forEach(p => ctx.lineTo(p.x, p.y));
  ctx.lineTo(pts[pts.length - 1].x, H - pad);
  ctx.closePath();
  ctx.fillStyle = grad;
  ctx.fill();

  // line
  ctx.beginPath();
  pts.forEach((p, i) => i === 0 ? ctx.moveTo(p.x, p.y) : ctx.lineTo(p.x, p.y));
  ctx.strokeStyle = '#00d4aa';
  ctx.lineWidth = 2;
  ctx.stroke();

  // last dot
  const last = pts[pts.length - 1];
  ctx.beginPath();
  ctx.arc(last.x, last.y, 4, 0, Math.PI * 2);
  ctx.fillStyle = '#00d4aa';
  ctx.fill();
}

function genPrices(base, n = 60) {
  const prices = [base];
  for (let i = 1; i < n; i++) {
    const delta = (Math.random() - 0.48) * base * 0.018;
    prices.push(Math.max(prices[i - 1] + delta, base * 0.7));
  }
  return prices;
}

    'Running ML prediction model...',
    'Analysing news sentiment...',
    'Building dashboard...'
  ];
  let si = 0;
  clearInterval(analysisTimer);
  analysisTimer = setInterval(() => {
    document.getElementById('loading-text').textContent = steps[Math.min(si++, steps.length - 1)];

  setTimeout(() => {
    clearInterval(analysisTimer);
    loader.classList.remove('visible');
    renderDashboard(raw);
    document.getElementById('dashboard').classList.add('visible');
  }, 2400);
}

  document.getElementById('pred-dn-pct').textContent = d.dnP;
  document.getElementById('conf-val').textContent = d.conf + '%';
  const confBar = document.getElementById('conf-bar');
  confBar.style.width = '0%';
  setTimeout(() => { confBar.style.width = d.conf + '%'; }, 200);
  const sigEl = document.getElementById('pred-signal');
  sigEl.textContent = d.sig;
  sigEl.style.color = d.sig.includes('BUY') ? 'var(--up)' : d.sig === 'SELL' ? 'var(--down)' : 'var(--text)';

  // Technical indicators
  const metricsEl = document.getElementById('tech-metrics');
  metricsEl.innerHTML = indicators(d).map(m => `
    <div>
      <div class="metric-row">
        <span class="mk">${m.k}</span>
        <span class="mv">${m.v}</span>
      </div>
      <div class="metric-bar-wrap">
        <div class="metric-bar-fill" style="width:${m.pct}%"></div>
      </div>
    </div>
  `).join('');

  // News
  const news = mockNews[sym] || mockNews.DEFAULT;
  document.getElementById('news-list').innerHTML = news.map(n => `
    <div class="news-item">
      <div class="news-dot ${n.s}"></div>
      <div>
        <div class="news-text">${n.t}</div>
        <div class="news-meta">${n.src} · ${n.time}</div>
      </div>
    </div>
  `).join('');
}
